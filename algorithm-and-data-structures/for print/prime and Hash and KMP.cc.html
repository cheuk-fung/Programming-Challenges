 2 {
 3     bool b[n];
 4     int sum = 0;
 5     for (int i = 0; i < n; i++) b[i] = true;
 6     for (int i = 2; i < n; i++) {
 7         if (b[i]) p[sum++] = i;
 8         for (int j = 0; j < sum && i * p[j] <= n; j++) {
 9             b[i * p[j]] = false;
10             if (i % p[j] == 0) break;
11         }
12     }
13 
14     return sum;
15 }

ELFHash

 1 int ELFHash(char* key)
 2 {
 3     unsigned int h = 0;
 4     while(*key)
 5     {
 6         h = (h << 4) + *key++;
 7         unsigned int g = h & 0Xf0000000;
 8         if (g) h ^= g >> 24;
 9         h &= ~g;
10     }
11     return h % MOD;
12 }

BKDRHash

2 {
3     unsigned int seed = 131; // 31 131 1313 13131 131313 etc..
4     unsigned int hash = 0;
5     while (*str)
6         hash = hash * seed + *str++;
7 
8     return (hash & 0x7FFFFFFF);
