11 #include <cstdio>
12 #include <cstring>
13 
14 int sum, p[100], cnt[100];
15 bool b[500];
16 
17 void prime()
18 {
19     memset(b, true, sizeof(b));
20     b[0] = b[1] = false;
21 
22     for (int i = 0; i < 500; i++) {
23         if (b[i]) p[sum++] = i;
24         for (int j = 0; j < sum && i * p[j] < 500; j++) {
25             b[i * p[j]] = false;
26             if (i % p[j] == 0) break;
27         }
28     }
29 }
30 
31 void divide(int n, int k)
32 {
33     memset(cnt, 0, sizeof(cnt));
34 
35     for (int i = 0; i < sum && n >= p[i]; i++) {
36         int a = n, b = k, c = n - k;
37 
38         while (a > 1) {
39             a /= p[i];
40             cnt[i] += a;
41         }
42 
43         while (b > 1) {
44             b /= p[i];
45             cnt[i] -= b;
46         }
47 
48         while (c > 1) {
49             c /= p[i];
50             cnt[i] -= c;
51         }
52     }
53 }
54 
55 bool solve()
56 {
57     int n, k;
58     if (scanf("%d%d", &n, &k) != 2) return false;
59     divide(n, k);
60 
61     long long ans = 1;
62     for (int i = 0; i < sum; i++)
63         if (cnt[i]) ans *= (cnt[i] + 1LL);
64 
65     printf("%lld\n", ans);
66 
67     return true;
68 }

70 int main()
71 {
72     prime();
73 
74     while (solve()) ;
75 
76     return 0;
