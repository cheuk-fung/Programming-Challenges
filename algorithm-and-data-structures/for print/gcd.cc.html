GCD

 1 int gcd(int a, int b)
 2 {
 3     if (b == 0) return a;
 4     return gcd(b, a % b);
 5 }

Extended Euclid

 7 int extendedEuclid(int a, int b, int* x, int* y)
 8 {
 9     if (b == 0) {
10         *x = 1;
11         *y = 0;
12 
13         return a;
14     }
15 
16     int r = extendedEuclid(b, a % b, x, y);
17     int t = *x;
18     *x = *y;
19     *y = t - a / b * *y;
20 
21     return r;
22 }

24 int lcm(int a, int b)
25 {
